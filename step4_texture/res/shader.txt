#shader vertex
#version 330 core
layout(location = 0) in vec3 pos;
layout(location = 1) in vec2 texcoord;
out vec2 TexCoord;
void main()
{
	gl_Position=vec4(pos,1.0f);
}

#shader fragment
#version 330 core
out vec4 FragColor;
in vec2 TexCoord;
uniform sampler2D texture;
void main()
{
	FragColor=texture(texture,TexCoord);
}
