#shader vertex
#version 330 core

layout(location=0) in vec3 vertexPos;

out vec3 pos;

void main()
{
	pos=vertexPos;
	gl_Position=vec4(vertexPos,1.0f);
}


#shader fragment

#version 330 core
in vec3 pos;
out vec4 FragColor;

uniform vec2 light_pos;

void main()
{
	float lightcolor=1/length(pos.xy-light_pos);
	FragColor=lightcolor*vec4(0.1f,0.1f,0.3f,1.0f);
}
