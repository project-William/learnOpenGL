#shader vertex
#version 330 core

layout(location=0) in vec3 vertexPos;

uniform mat4 model;
uniform mat4 view;
uniform mat4 projection;

void main()
{

	gl_Position= projection*view*model*vec4(vertexPos,1.0f);
}


#shader fragment

#version 330 core

out vec4 FragColor;

uniform vec2 light_pos;
void main()
{
	
	FragColor=vec4(0.1f,0.1f,0.3f,1.0f);
}
